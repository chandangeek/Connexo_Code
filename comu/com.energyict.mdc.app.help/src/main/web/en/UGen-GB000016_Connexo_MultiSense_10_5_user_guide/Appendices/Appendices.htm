Calendar format

Time of use calendar format

When you import a time of use calendar in MultiSense the XML file should have the following format:

<?xml version="1.0" encoding="UTF-8"?>

<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema" elementFormDefault="qualified">

 <xs:element name="calendar">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="content-version"/>

       <xs:element ref="name"/>

       <xs:element ref="category"/>

       <xs:element ref="" minOccurs="0"/>

       <xs:element ref="description" minOccurs="0"/>

       <xs:element ref="timezone"/>

       <xs:element ref="start-year"/>

       <xs:element ref="events"/>

       <xs:element ref="day-types"/>

       <xs:element ref="periods"/>

       <xs:element ref="exceptions"/>

       <xs:any minOccurs="0"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="content-version" type="xs:NMTOKEN"/>

 <xs:element name="category" type="xs:NCName"/>

 <xs:element name="" type="xs:NCName"/>

 <xs:element name="description" type="xs:string"/>

 <xs:element name="timezone" type="xs:string"/>

 <xs:element name="start-year" type="xs:integer"/>

 <xs:element name="events">

   <xs:complexType>

     <xs:sequence>

       <xs:element maxOccurs="unbounded" ref="event"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="day-types">

   <xs:complexType>

     <xs:sequence>

       <xs:element maxOccurs="unbounded" ref="day-type"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="day-type">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="id"/>

       <xs:element ref="name"/>

       <xs:element ref="ranges"/>

       <xs:any minOccurs="0"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="ranges">

   <xs:complexType>

     <xs:sequence>

       <xs:element maxOccurs="unbounded" ref="range-time"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="range-time">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="from"/>

       <xs:element name="event" type="xs:integer"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="from">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="hour"/>

       <xs:element ref="minute"/>

       <xs:element ref="second"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="hour" type="xs:integer"/>

 <xs:element name="minute" type="xs:integer"/>

 <xs:element name="second" type="xs:integer"/>

 <xs:element name="periods">

   <xs:complexType>

     <xs:sequence>

       <xs:element maxOccurs="unbounded" ref="period"/>

       <xs:element ref="transitions"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="period">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="id"/>

       <xs:element ref="name"/>

       <xs:element ref="week-template"/>

       <xs:any minOccurs="0"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="week-template">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="monday"/>

       <xs:element ref="tuesday"/>

       <xs:element ref="wednesday"/>

       <xs:element ref="thursday"/>

       <xs:element ref="friday"/>

       <xs:element ref="saturday"/>

       <xs:element ref="sunday"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="monday">

   <xs:complexType>

     <xs:attribute name="day-type" use="required" type="xs:integer"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="tuesday">

   <xs:complexType>

     <xs:attribute name="day-type" use="required" type="xs:integer"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="wednesday">

   <xs:complexType>

     <xs:attribute name="day-type" use="required" type="xs:integer"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="thursday">

   <xs:complexType>

     <xs:attribute name="day-type" use="required" type="xs:integer"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="friday">

   <xs:complexType>

     <xs:attribute name="day-type" use="required" type="xs:integer"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="saturday">

   <xs:complexType>

     <xs:attribute name="day-type" use="required" type="xs:integer"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="sunday">

   <xs:complexType>

     <xs:attribute name="day-type" use="required" type="xs:integer"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="transitions">

   <xs:complexType>

     <xs:sequence>

       <xs:element maxOccurs="unbounded" ref="transition"/>

     </xs:sequence>

     <xs:attribute name="recurring" use="required" type="xs:boolean"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="transition">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="to-period"/>

       <xs:element ref="year" minOccurs="0"/>

       <xs:element ref="month"/>

       <xs:element ref="day"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="to-period" type="xs:integer"/>

 <xs:element name="exceptions">

   <xs:complexType>

     <xs:sequence>

       <xs:element minOccurs="0" maxOccurs="unbounded" ref="exception"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="exception">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="occurrences"/>

     </xs:sequence>

     <xs:attribute name="day-type" use="required" type="xs:integer"/>

   </xs:complexType>

 </xs:element>

 <xs:element name="occurrences">

   <xs:complexType>

     <xs:choice maxOccurs="unbounded">

       <xs:element ref="fixed-occurrence"/>

       <xs:element ref="recurring-occurrence"/>

     </xs:choice>

   </xs:complexType>

 </xs:element>

 <xs:element name="recurring-occurrence">

   <xs:complexType>

     <xs:sequence>

       <xs:element ref="month"/>

       <xs:element ref="day"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="name" type="xs:string"/>

 <xs:element name="event">

   <xs:complexType mixed="true">

     <xs:sequence>

       <xs:element ref="id"/>

       <xs:element ref="name"/>

       <xs:element ref="code"/>

       <xs:any minOccurs="0"/>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="code" type="xs:integer"/>

 <xs:element name="id" type="xs:integer"/>

 <xs:element name="month" type="xs:integer"/>

 <xs:element name="day" type="xs:integer"/>

 <xs:element name="fixed-occurrence">

   <xs:complexType>

     <xs:sequence>

       <xs:element minOccurs="0" ref="fixed-occurrence"/>

       <xs:sequence minOccurs="0">

         <xs:element ref="day"/>

         <xs:element ref="month"/>

         <xs:element ref="year"/>

       </xs:sequence>

     </xs:sequence>

   </xs:complexType>

 </xs:element>

 <xs:element name="year" type="xs:integer"/>

</xs:schema>

Time of use calendar example

Below you can find an example of an XML file of a time of use calendar. This complies to the scheme discussed above.

<?xml version="1.0" encoding="UTF-8"?>

<calendar>

   <content-version>1.0.0</content-version>

   <name>Residential TOU Example</name>

   <category>TOU</category>

   <>optional</>

   <description>Example calendar</description>

   <timezone>GMT</timezone>

   <start-year>2010</start-year>

   <events>

       <event>

           <id>1</id>

           <name>On Peak</name>

           <code>3</code>

       </event>

       <event>

           <id>2</id>

           <name>Off Peak</name>

           <code>5</code>

       </event>

       <event>

           <id>3</id>

           <name>Demand response</name>

           <code>97</code>

       </event>

   </events>

   <day-types>

       <day-type>

           <id>1</id>

           <name>summer weekday</name>

           <ranges>

               <range-time>

                   <!-- Closed-Open -->

                   <from>

                       <!-- maps to java class java.time.LocalTime -->

                       <hour>0</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>2</event>

               </range-time>

               <range-time>

                   <from>

                       <hour>13</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>1</event>

               </range-time>

               <range-time>

                   <from>

                       <hour>20</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>2</event>

               </range-time>

           </ranges>

       </day-type>

       <day-type>

           <id>2</id>

           <name>weekend</name>

           <ranges>

               <range-time>

                   <from>

                       <hour>0</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>2</event>

               </range-time>

           </ranges>

       </day-type>

       <day-type>

           <id>3</id>

           <name>holiday</name>

           <ranges>

               <range-time>

                   <from>

                       <hour>0</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>2</event>

               </range-time>

           </ranges>

       </day-type>

       <day-type>

           <id>4</id>

           <name>winter day</name>

           <ranges>

               <range-time>

                   <!-- Closed-Open -->

                   <from>

                       <!-- maps to java class java.time.LocalTime -->

                       <hour>0</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>2</event>

               </range-time>

               <range-time>

                   <from>

                       <hour>5</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>1</event>

               </range-time>

               <range-time>

                   <from>

                       <hour>9</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>1</event>

               </range-time>

               <range-time>

                   <from>

                       <hour>17</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>2</event>

               </range-time>

               <range-time>

                   <from>

                       <hour>21</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>2</event>

               </range-time>

           </ranges>

       </day-type>

       <day-type>

           <id>5</id>

           <name>Demand Response</name>

           <ranges>

               <range-time>

                   <from>

                       <hour>0</hour>

                       <minute>0</minute>

                       <second>0</second>

                   </from>

                   <event>3</event>

               </range-time>

           </ranges>

       </day-type>

   </day-types>

   <periods>

       <period>

           <id>1</id>

           <name>Summer</name>

           <week-template>

               <monday day-type="1"/>

               <tuesday day-type="1"/>

               <wednesday day-type="1"/>

               <thursday day-type="1"/>

               <friday day-type="1"/>

               <saturday day-type="2"/>

               <sunday day-type="2"/>

           </week-template>

       </period>

       <period>

           <id>2</id>

           <name>Winter</name>

           <week-template>

               <monday day-type="4"/>

               <tuesday day-type="4"/>

               <wednesday day-type="4"/>

               <thursday day-type="4"/>

               <friday day-type="4"/>

               <saturday day-type="4"/>

               <sunday day-type="4"/>

           </week-template>

       </period>

       <transitions recurring="true">

           <!-- When recurring is true, all transition elements

                will only have month/day and not year

                and will map to a java.util.MonthDay

                When recurring is false, all transition elements

                will have year/month/day and will map to java.util.LocalDate -->

           <transition>

               <to-period>1</to-period>

               <month>5</month>

               <day>1</day>

           </transition>

           <transition>

               <to-period>2</to-period>

               <month>11</month>

               <day>1</day>

           </transition>

       </transitions>

   </periods>

   <exceptions>

       <exception day-type="3">

           <occurrences>

               <recurring-occurrence>

                   <!-- maps to java class java.time.MonthDay -->

                   <month>1</month>

                   <day>1</day>

               </recurring-occurrence>

               <fixed-occurrence>

                   <day>18</day>

                   <month>1</month>

                   <year>2016</year>

               </fixed-occurrence>

               <fixed-occurrence>

                   <day>15</day>

                   <month>2</month>

                   <year>2016</year>

               </fixed-occurrence>

               <fixed-occurrence>

                   <day>30</day>

                   <month>5</month>

                   <year>2016</year>

               </fixed-occurrence>

               <recurring-occurrence>

                   <!-- maps to java class java.time.MonthDay -->

                   <month>7</month>

                   <day>4</day>

               </recurring-occurrence>

               <fixed-occurrence>

                   <day>5</day>

                   <month>9</month>

                   <year>2016</year>

               </fixed-occurrence>

               <fixed-occurrence>

                   <day>10</day>

                   <month>10</month>

                   <year>2016</year>

               </fixed-occurrence>

               <recurring-occurrence>

                   <!-- maps to java class java.time.MonthDay -->

                   <month>11</month>

                   <day>11</day>

               </recurring-occurrence>

               <fixed-occurrence>

                   <day>24</day>

                   <month>11</month>

                   <year>2016</year>

               </fixed-occurrence>

               <recurring-occurrence>

                   <!-- maps to java class java.time.MonthDay -->

                   <month>12</month>

                   <day>25</day>

               </recurring-occurrence>

               <recurring-occurrence>

                   <!-- maps to java class java.time.MonthDay -->

                   <month>12</month>

                   <day>26</day>

               </recurring-occurrence>

           </occurrences>

       </exception>

       <exception day-type="5">

           <occurrences>

               <fixed-occurrence>

                   <day>20</day>

                   <month>5</month>

                   <year>2016</year>

               </fixed-occurrence>

           </occurrences>

       </exception>

   </exceptions>

</calendar>
